// BEFORE (Express/Multer):
import { diskStorage } from "multer";
import { FileInterceptor } from "@nestjs/platform-express";

@Post("upload")
@UseInterceptors(
  FileInterceptor("file", {
    storage: diskStorage({
      destination: "./uploads/employees",
      filename: (req, file, cb) => {
        const uniqueSuffix = Date.now() + "-" + Math.round(Math.random() * 1e9);
        cb(null, file.fieldname + "-" + uniqueSuffix + ".xlsx");
      },
    }),
  })
)
async uploadFile(
  @UploadedFile() file: Express.Multer.File,
  @Request() req
) {
  return this.service.processUpload(file);
}

// AFTER (Fastify):
import { createWriteStream } from "fs";
import { pipeline } from "stream/promises";
import { join } from "path";

@Post("upload")
async uploadFile(@Request() req) {
  const data = await req.file(); // Get file from Fastify multipart

  if (!data) {
    throw new BadRequestException("No file uploaded");
  }

  // Save file manually
  const uploadPath = join(process.cwd(), "uploads", "employees");
  const uniqueSuffix = Date.now() + "-" + Math.round(Math.random() * 1e9);
  const filename = `file-${uniqueSuffix}.xlsx`;
  const filepath = join(uploadPath, filename);

  // Write file to disk
  await pipeline(data.file, createWriteStream(filepath));

  // Create file object similar to Multer
  const file = {
    filename,
    path: filepath,
    mimetype: data.mimetype,
    size: data.file.bytesRead,
  };

  return this.service.processUpload(file);
}

// ALTERNATIVE: Create a Fastify File Interceptor wrapper
// Create: src/interceptors/fastify-file.interceptor.ts

import {
  Injectable,
  NestInterceptor,
  ExecutionContext,
  CallHandler,
  BadRequestException,
} from "@nestjs/common";
import { Observable } from "rxjs";
import { createWriteStream, mkdirSync, existsSync } from "fs";
import { pipeline } from "stream/promises";
import { join } from "path";

export interface FastifyFileInterceptorOptions {
  destination: string;
  filename?: (file: any) => string;
}

export function FastifyFileInterceptor(
  fieldName: string,
  options: FastifyFileInterceptorOptions
) {
  @Injectable()
  class MixinInterceptor implements NestInterceptor {
    async intercept(
      context: ExecutionContext,
      next: CallHandler
    ): Promise<Observable<any>> {
      const request = context.switchToHttp().getRequest();
      const data = await request.file();

      if (!data) {
        throw new BadRequestException("No file uploaded");
      }

      // Ensure upload directory exists
      if (!existsSync(options.destination)) {
        mkdirSync(options.destination, { recursive: true });
      }

      // Generate filename
      const filename = options.filename
        ? options.filename(data)
        : `${Date.now()}-${Math.round(Math.random() * 1e9)}.${data.mimetype.split("/")[1]}`;

      const filepath = join(options.destination, filename);

      // Write file
      await pipeline(data.file, createWriteStream(filepath));

      // Attach file info to request (similar to Multer)
      request.file = {
        fieldname: data.fieldname,
        originalname: data.filename,
        encoding: data.encoding,
        mimetype: data.mimetype,
        filename,
        path: filepath,
        destination: options.destination,
      };

      return next.handle();
    }
  }
  return MixinInterceptor;
}
